      function setSpan(){

        if(i>=5)
          i=1;
        else
          i++;
        var Image=i+".jpg";
        img.src=Image;
      }
      function stopRotate(){
        count++;
        if(count%2==1)
          clearInterval(interval);
        else {
          setSpan();
        }
      }
